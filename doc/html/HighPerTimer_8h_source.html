574 inline HighPerTimer operator+ ( const HighPerTimer & Timer1, const HighPerTimer & Timer2 )
575 {
576  if ( ( Timer2.HPTics() > 0 ) && ( ( HighPerTimer::HPTimer_MAX.HPTics() - ( Timer2.HPTics() ) ) <= ( Timer1.HPTics() ) ) )
577  {
578  throw std::out_of_range ( "HPTimer overflow" );
579  }
580  if ( ( Timer2.HPTics() < 0 ) && ( ( HighPerTimer::HPTimer_MIN.HPTics() - ( Timer2.HPTics() ) ) >= ( Timer1.HPTics() ) ) )
581  {
582  throw std::out_of_range ( "HPTimer overflow" );
583  }
584  return HighPerTimer { ( Timer1.HPTics() + Timer2.HPTics() ), false };
585 }
586 
593 inline HighPerTimer operator- ( const HighPerTimer & Timer1, const HighPerTimer & Timer2 )
594 {
595  if ( ( Timer2.HPTics() > 0 ) && ( ( HighPerTimer::HPTimer_MIN.HPTics() + ( Timer2.HPTics() ) ) >= ( Timer1.HPTics() ) ) )
596  {
597  throw std::out_of_range ( "HPTimer overflow" );
598  }
599  if ( ( Timer2.HPTics() < 0 ) && ( ( HighPerTimer::HPTimer_MAX.HPTics() + ( Timer2.HPTics() ) ) <= ( Timer1.HPTics() ) ) )
600  {
601  throw std::out_of_range ( "HPTimer overflow" );
602  }
603  return HighPerTimer { ( Timer1.HPTics() - Timer2.HPTics() ), false };
604 }
